Re-create the figure (static)

The original figure is a share-of-members-by-generation plot across Congresses. We compute the proportion of members in each generation for each Congress.

gen_levels <- c(
  "Gilded", "Progressive", "Missionary", "Lost", "Greatest", "Silent",
  "Boomers", "Gen X", "Millennial", "Gen Z"
)

plot_df <- dat %>%
  mutate(
    generation = case_when(
      generation == "baby boomer" ~ "Boomers",
      generation == "Generation X" ~ "Gen X",
      generation == "millennial" ~ "Millennial",
      generation == "Generation Z" ~ "Gen Z",
      TRUE ~ generation
    ),
    generation = factor(generation, levels = gen_levels),
    start_date = as.Date(start_date)
  ) %>%
  filter(!is.na(generation), !is.na(start_date)) %>%
  count(congress, start_date, generation, name = "n") %>%
  group_by(congress, start_date) %>%
  mutate(share = n / sum(n)) %>%
  ungroup()

p <- ggplot(plot_df, aes(x = start_date, y = share, fill = generation)) +
  geom_area(position = "fill", linewidth = 0) +
  scale_y_continuous(labels = percent_format(accuracy = 1)) +
  scale_x_date(
    date_breaks = "5 years",

if (!"prop" %in% names(plot_df)) {
  plot_df <- plot_df %>%
    dplyr::group_by(congress, start_date) %>%

tbl_df <- plot_df %>%
  dplyr::select(congress, start_date, generation, prop) %>%
  tidyr::pivot_wider(names_from = generation, values_from = prop) %>%
  dplyr::arrange(dplyr::desc(start_date)) %>%
  dplyr::mutate(
    Year = lubridate::year(start_date),
    Boomers_change_pp = (Boomers - dplyr::lead(Boomers)) * 100
  ) %>%
  dplyr::slice(1:10) %>%
  dplyr::select(Year, congress, Boomers, Boomers_change_pp, `Gen X`, Millennial, `Gen Z`)
